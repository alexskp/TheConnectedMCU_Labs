182 /*** void loop()
183  *
184  * Parameters:

186  *
187  * Return Values:
188  * None
189  *
190  * Description:
191  *
192  * Arduino loop function.
193  *
194  * We are using the default timeout values for the DEIPcK and TcpClient class

196  *
197  * This code will write some stings to the server and have the server echo it back
198  *
199  * ------------------------------------------------------------ */
200 void loop() {
201  IPSTATUS status;
202  int cbRead = 0;
203 
204  switch(state)
205  {
206 
207  case CONNECT:
208  if(WiFiConnectMacro())
209  {
210 // Serial.println("WiFi connected");
211  display.setCursor(0,10);
212  display.print("Connected to ");
213  display.println(szSsid);
214  display.display();
215  delay(2000);
216  //display.clearDisplay();
217  deIPcK.begin();
218  state = TCPCONNECT;
219  }
220  else if(IsIPStatusAnError(status))
221  {
222 // Serial.print("Unable to connection, status: ");
223 // Serial.println(status, DEC);
224  display.setCursor(0,10);
225  display.println("Unable to connection, status: ");
226  display.println(status, DEC);
227  display.display();
228  delay(500);
229  state = CLOSE;
230  }
231  break;
232 
233  case TCPCONNECT:
234  //Serial.println("____");
235  if(deIPcK.tcpConnect(szIPServer, portServer, tcpSocket))
236  {
237 // Serial.println("Connected to server.");
238  display.setCursor(0,20);
239  display.println("Loading weather from server");
240  display.display();
241  delay(2000);
242  state = WRITE;
243  }
244  break;
245 
246  // write out the strings
247  case WRITE:
248  if(tcpSocket.isEstablished())
249  {
250  tcpSocket.writeStream(rgbWriteStream, cbWriteStream);
251 
252 // Serial.println("Bytes Read Back:");
253  state = READ;
254  tStart = (unsigned) millis();
255  }
256  break;
257 
258  // look for the echo back
259  case READ:
260  // see if we got anything to read
261  if((cbRead = tcpSocket.available()) > 0)
262  {
263  cbRead = cbRead < (int) sizeof(rgbRead) ? cbRead : sizeof(rgbRead);
264  cbRead = tcpSocket.readStream(rgbRead, cbRead);
265 
266 // for(int i=0; i < cbRead; i++)
267 // {
268 // Serial.print((char) rgbRead[i]);
269 // }
270  json_parse();
271  displ_weather();
272  }
273 
274  // give us some time to get everything echo'ed back
275  else if( (((unsigned) millis()) - tStart) > tWait )
276  {
277 // Serial.println("");
278  state = CLOSE;
279  }
280  break;
281 
282  // done, so close up the tcpSocket
283  case CLOSE:
284  tcpSocket.close();
285 // Serial.println("Closing TcpClient, Done with sketch.");
286  state = TCPCONNECT;
287  break;
288 
289  case DONE:
290  default:
291  break;
292  }
293 
294  // keep the stack alive each pass through the loop()
295  DEIPcK::periodicTasks();
296 }

350  // StaticJsonBuffer allocates memory on the stack, it can be
351  // replaced by DynamicJsonBuffer which allocates in the heap.

353  // DynamicJsonBuffer jsonBuffer(200);
354 
355  // JSON input string.
356  //
357  // It's better to use a char[] as shown here.
358  // If you use a const char* or a String, ArduinoJson will
359  // have to make a copy of the input in the JsonBuffer.
360  //char json[] =
361  // "{\"sensor\":\"gps\",\"time\":1351824120,\"data\":[48.756080,2.302038]}";
