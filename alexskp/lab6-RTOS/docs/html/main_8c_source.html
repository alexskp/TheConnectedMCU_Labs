251 /* This function overrides the normal _weak_ generic handler. */
252 void _general_exception_handler(void)
253 {
254 static enum {
255  EXCEP_IRQ = 0, /* interrupt */
256  EXCEP_AdEL = 4, /* address error exception (load or ifetch) */
257  EXCEP_AdES, /* address error exception (store) */
258  EXCEP_IBE, /* bus error (ifetch) */
259  EXCEP_DBE, /* bus error (load/store) */
260  EXCEP_Sys, /* syscall */
261  EXCEP_Bp, /* breakpoint */
262  EXCEP_RI, /* reserved instruction */
263  EXCEP_CpU, /* coprocessor unusable */
264  EXCEP_Overflow, /* arithmetic overflow */
265  EXCEP_Trap, /* trap (possible divide by zero) */
266  EXCEP_FPE = 15, /* floating point exception */
267  EXCEP_IS1 = 16, /* implementation specfic 1 */
268  EXCEP_CEU, /* CorExtend Unuseable */
269  EXCEP_C2E, /* coprocessor 2 */
270  EXCEP_DSPDis = 26 /* DSP module disabled */
271 } _excep_code;
272 
273 static unsigned long _epc_code;
274 static unsigned long _excep_addr;
275 
276  asm volatile( "mfc0 %0,$13" : "=r" (_epc_code) );
277  asm volatile( "mfc0 %0,$14" : "=r" (_excep_addr) );
278 
279  _excep_code = ( _epc_code & 0x0000007C ) >> 2;
280 
281  for( ;; )
282  {
283  /* prevent compiler warning */
284  (void) _excep_code;
285 
286  /* Examine _excep_code to identify the type of exception. Examine
287  _excep_addr to find the address that caused the exception */
288  Nop();
289  Nop();
290  Nop();
291  }
292 }
