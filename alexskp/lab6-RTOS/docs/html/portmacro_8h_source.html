81 /* Critical section management. */
82 #define portIPL_SHIFT ( 10UL )
83 /* Don't straddle the CEE bit. Interrupts calling FreeRTOS functions should
84 never have higher IPL bits set anyway. */
85 #define portALL_IPL_BITS ( 0x7FUL << portIPL_SHIFT )
86 #define portSW0_BIT ( 0x01 << 8 )
87 
88 /* This clears the IPL bits, then sets them to
89 configMAX_SYSCALL_INTERRUPT_PRIORITY. An extra check is performed if
90 configASSERT() is defined to ensure an assertion handler does not inadvertently
91 attempt to lower the IPL when the call to assert was triggered because the IPL
92 value was found to be above configMAX_SYSCALL_INTERRUPT_PRIORITY when an ISR
93 safe FreeRTOS API function was executed. ISR safe FreeRTOS API functions are
94 those that end in FromISR. FreeRTOS maintains a separate interrupt API to
95 ensure API function and interrupt entry is as fast and as simple as possible. */
96 #ifdef configASSERT
97  #define portDISABLE_INTERRUPTS() \
98  { \
99  uint32_t ulStatus; \
100  \
101  /* Mask interrupts at and below the kernel interrupt priority. */ \
102  ulStatus = _CP0_GET_STATUS(); \
103  \
104  /* Is the current IPL below configMAX_SYSCALL_INTERRUPT_PRIORITY? */ \
105  if( ( ( ulStatus & portALL_IPL_BITS ) >> portIPL_SHIFT ) < configMAX_SYSCALL_INTERRUPT_PRIORITY ) \
106  { \
107  ulStatus &= ~portALL_IPL_BITS; \
108  _CP0_SET_STATUS( ( ulStatus | ( configMAX_SYSCALL_INTERRUPT_PRIORITY << portIPL_SHIFT ) ) ); \
109  } \
110  }
111 #else /* configASSERT */
112  #define portDISABLE_INTERRUPTS() \
113  { \
114  uint32_t ulStatus; \
115  \
116  /* Mask interrupts at and below the kernel interrupt priority. */ \
117  ulStatus = _CP0_GET_STATUS(); \
118  ulStatus &= ~portALL_IPL_BITS; \
119  _CP0_SET_STATUS( ( ulStatus | ( configMAX_SYSCALL_INTERRUPT_PRIORITY << portIPL_SHIFT ) ) ); \
120  }
121 #endif /* configASSERT */

128  ulStatus = _CP0_GET_STATUS(); \
